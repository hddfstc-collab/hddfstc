const unreadPopupTargets = useMemo(() => { return (dynamicApps || []).filter(app => { if (['form', 'reservation'].includes(app.actionType)) return false; // 1. 공지사항 위젯 체크 (첫 번째 코드의 장점) if (app.actionType === 'widget_notices') { return (notices || []).some(n => !readNoticeIds.includes(n.id)); } // 2. 일반 게시판 체크 (두 번째 코드의 방어 로직 적용) return (dynamicPosts || []).some(p => { if (p.appId !== app.id) return false; const postDate = new Date(p.createdAt); // 날짜 형식이 올바르지 않으면 NaN이 나올 수 있으므로 체크 const isNew = (new Date() - postDate) / (1000 * 60 * 60) <= 72; const isUnread = !readPostIds.includes(p.id); return isNew && isUnread; }); }); }, [dynamicApps, notices, dynamicPosts, readNoticeIds, readPostIds]);
